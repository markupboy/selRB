  def test_global_links
     global_links = [
      { :locator => 'link=Idea Hub', :page_title => 'Idea Hub :: American Express OPEN Forum'  }
    ]
    global_links.each { |link|
      browser.open "/"
      assertElementPresent link[:locator]
      browser.click link[:locator], :wait_for => :page
      verifyTitle link[:page_title]
    }
  end
