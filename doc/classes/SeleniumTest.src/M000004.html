  def setup
    @verification_errors = []
    @browser = Selenium::Client::Driver.new \
        :host => TEST_CONFIG['host'],
        :port => TEST_CONFIG['port'],
        :browser => "*firefox",
        :url => TEST_CONFIG['url'],
        :timeout_in_second => TEST_CONFIG['timeout']
        
    browser.start_new_browser_session
  end
